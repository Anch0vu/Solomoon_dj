#!/usr/bin/liquidsoap
#
# SoloMoon's Club — Phase 1 + 3: 24/7 playlist + Live DJ via harbor
# Liquidsoap 2.2.x
#
# Поток данных:
#   browser (Web Audio API) → MediaRecorder → WebSocket
#   → radio-api → ffmpeg → input.harbor:8005
#   → fallback([live, playlist]) → output.icecast /stream.mp3

# ============== settings ==============
settings.log.level.set(3)
settings.log.stdout.set(true)
settings.server.telnet.set(true)
# Bind only to loopback — telnet must not be reachable from other containers.
settings.server.telnet.bind_addr.set("127.0.0.1")
settings.server.telnet.port.set(1234)

# ============== live DJ input (harbor) ==============
# Liquidsoap слушает на :8005, ждёт Icecast-source-protocol подключения
# от ffmpeg (через radio-api). Как только источник подключается —
# harbor становится доступен и вытесняет плейлист через fallback ниже.
live = input.harbor(
  id="live",
  port=8005,
  auth=fun(login, password) ->
    password == environment.get("HARBOR_PASSWORD"),
  "/live"
)

# ============== playlist source ==============
# reload каждые 30 минут — успеешь дозалить треки rsync'ом
playlist_source = playlist(
  id="fallback",
  mode="randomize",
  reload=1800,
  reload_mode="rounds",
  "/music"
)

# Если папка пустая — бесконечная тишина, чтобы liquidsoap не падал.
# blank() — единственный бесконечный молчащий источник; sine с duration= конечен.
silence = blank()
safe_source = fallback(track_sensitive=false, [playlist_source, silence])

# Кроссфейд применяем ТОЛЬКО к плейлисту — DJ сам микширует в браузере,
# normalize тоже только на плейлист (не мешает задуманным фейдам DJ-а).
playlist_mixed = crossfade(
  duration=3.0,
  fade_in=2.0,
  fade_out=2.0,
  smart=true,
  safe_source
)

playlist_normalized = normalize(
  target=-14.0,
  gain_min=-6.0,
  gain_max=6.0,
  playlist_mixed
)

# ============== final source ==============
# live (harbor) имеет приоритет; как только DJ выходит — возвращается плейлист.
radio = fallback(track_sensitive=false, [live, playlist_normalized])

# ============== Icecast output ==============
output.icecast(
  %mp3(bitrate=320, samplerate=44100, stereo=true),
  id="solomoon-stream",
  host=environment.get("ICECAST_HOST"),
  port=int_of_string(environment.get("ICECAST_PORT")),
  password=environment.get("ICECAST_SOURCE_PASSWORD"),
  mount="/stream.mp3",
  name="SoloMoon's Club",
  description="House · Deep House · EDM · R&B — Доброград",
  genre="House Deep-House EDM R&B",
  url="http://95.165.172.153:8000/",
  public=false,
  encoding="UTF-8",
  chunked=true,
  send_icy_metadata=true,
  # max=5.0 ограничивает буфер 5-ю секундами — без max буфер растёт неограниченно
  # если Icecast потребляет медленнее, чем Liquidsoap производит (memory leak).
  buffer(buffer=0.2, max=5.0, radio)
)

log("=== SoloMoon's Club is on air ===")
